    m = 4294967296;

        this.x = seed
    }

        return this.x;
    }

fixed FPS

                                    
class Loop {
    constructor(fps, callback) {
        this.oneFrameMs = 1000 / fps;
        this.callback = callback;
        this.msCounter = 0;
    }

    play(delta) {
        this.msCounter += delta;
        if(this.msCounter > this.oneFrameMs){
            this.callback();
            this.msCounter = this.msCounter % this.oneFrameMs;
        }
    }
}
